### Task 1: Create Optimize API Endpoint

**File**: `src/app/api/creator-workflows/[id]/optimize/route.ts`

This endpoint:
- Fetches workflow data (niche, script from Create phase)
- Calls `runPredictionPipeline`
- Saves results as artifact
- Returns pack results for UI

```typescript
import { NextRequest, NextResponse } from 'next/server';
import { createClient } from '@/lib/supabase/server';
import { runPredictionPipeline } from '@/lib/prediction/runPredictionPipeline';

interface RouteContext {
  params: Promise<{ id: string }>;
}

// POST /api/creator-workflows/[id]/optimize - Run prediction
export async function POST(request: NextRequest, context: RouteContext) {
  try {
    const { id: workflowId } = await context.params;
    const supabase = await createClient();
    const { data: { user }, error: authError } = await supabase.auth.getUser();

    if (authError || !user) {
      return NextResponse.json(
        { data: null, error: 'Unauthorized' },
        { status: 401 }
      );
    }

    // Verify workflow ownership
    const { data: workflow, error: workflowError } = await supabase
      .from('workflow_runs')
      .select('id, metadata')
      .eq('id', workflowId)
      .eq('user_id', user.id)
      .single();

    if (workflowError || !workflow) {
      return NextResponse.json(
        { data: null, error: 'Workflow not found' },
        { status: 404 }
      );
    }

    // Fetch step data (especially Create phase for script/video)
    const { data: steps, error: stepsError } = await supabase
      .from('workflow_run_steps')
      .select('phase_number, input_data')
      .eq('workflow_run_id', workflowId);

    if (stepsError) throw stepsError;

    // Extract relevant data from steps
    const researchStep = steps?.find(s => s.phase_number === 1);
    const createStep = steps?.find(s => s.phase_number === 3);

    const niche = researchStep?.input_data?.niche || workflow.metadata?.niche || 'general';
    const script = createStep?.input_data?.videoDetails?.description || '';
    const title = createStep?.input_data?.videoDetails?.title || '';

    // Build transcript from available data
    const transcript = script || title || 'No content available for analysis';

    if (transcript === 'No content available for analysis') {
      return NextResponse.json(
        { data: null, error: 'Please add video content in the Create phase before optimizing' },
        { status: 400 }
      );
    }

    // Run prediction pipeline
    const pipelineResult = await runPredictionPipeline({
      transcript,
      niche,
      mode: 'pattern_based', // Use full pack analysis
      videoId: workflowId, // Use workflow ID as video ID for tracking
      metadata: {
        source: 'creator_workflow',
        workflow_id: workflowId,
      },
    });

    // Extract pack results
    const qualitativeAnalysis = pipelineResult.qualitative_analysis || {};
    const pack1 = qualitativeAnalysis.pack1 || null;
    const pack2 = qualitativeAnalysis.pack2 || null;
    const pack3 = qualitativeAnalysis.pack3 || null;
    const packV = qualitativeAnalysis.packV || null;

    // Calculate overall DPS (from pipeline or compute from packs)
    const dpsScore = pipelineResult.predicted_dps_7d || calculateDPSFromPacks(pack1, pack3);

    // Generate Gate A warnings based on pack results
    const gateAWarnings = generateGateAWarnings(pack1, pack3);

    // Save prediction result as artifact
    const { error: artifactError } = await supabase
      .from('workflow_run_artifacts')
      .insert({
        workflow_run_id: workflowId,
        artifact_type: 'prediction_result',
        artifact_data: {
          dps_score: dpsScore,
          pack1,
          pack2,
          pack3,
          packV,
          gate_a_warnings: gateAWarnings,
          run_id: pipelineResult.run_id,
          predicted_at: new Date().toISOString(),
        },
        version: 1,
      });

    if (artifactError) {
      console.error('Failed to save prediction artifact:', artifactError);
      // Don't fail the request, just log
    }

    // Update Optimize step with gate check results
    await supabase
      .from('workflow_run_steps')
      .update({
        output_data: {
          dps_score: dpsScore,
          pack1_summary: pack1 ? summarizePack1(pack1) : null,
          pack2_suggestions: pack2?.changes || [],
          gate_a_warnings: gateAWarnings,
        },
        gate_check_results: {
          hook_effectiveness: !gateAWarnings.some(w => w.type === 'hook'),
          proof_quality: !gateAWarnings.some(w => w.type === 'proof'),
          cta_alignment: !gateAWarnings.some(w => w.type === 'cta'),
          warnings: gateAWarnings,
        },
      })
      .eq('workflow_run_id', workflowId)
      .eq('phase_number', 4);

    // Update workflow_performance with predicted metrics
    await supabase
      .from('workflow_performance')
      .upsert({
        workflow_run_id: workflowId,
        predicted_dps: dpsScore,
        predicted_engagement: pack1?.grader_confidence || 0.5,
      }, {
        onConflict: 'workflow_run_id',
      });

    return NextResponse.json({
      data: {
        dps_score: dpsScore,
        pack1,
        pack2,
        pack3,
        packV,
        gate_a_warnings: gateAWarnings,
        run_id: pipelineResult.run_id,
      },
      error: null,
    });

  } catch (err) {
    console.error('Error in POST /api/creator-workflows/[id]/optimize:', err);
    return NextResponse.json(
      { data: null, error: 'Prediction failed. Please try again.' },
      { status: 500 }
    );
  }
}

// GET /api/creator-workflows/[id]/optimize - Get latest prediction result
export async function GET(request: NextRequest, context: RouteContext) {
  try {
    const { id: workflowId } = await context.params;
    const supabase = await createClient();
    const { data: { user }, error: authError } = await supabase.auth.getUser();

    if (authError || !user) {
      return NextResponse.json(
        { data: null, error: 'Unauthorized' },
        { status: 401 }
      );
    }

    // Get latest prediction artifact
    const { data: artifact, error: artifactError } = await supabase
      .from('workflow_run_artifacts')
      .select('artifact_data, created_at')
      .eq('workflow_run_id', workflowId)
      .eq('artifact_type', 'prediction_result')
      .order('created_at', { ascending: false })
      .limit(1)
      .single();

    if (artifactError) {
      if (artifactError.code === 'PGRST116') {
        return NextResponse.json(
          { data: null, error: 'No prediction found. Run analysis first.' },
          { status: 404 }
        );
      }
      throw artifactError;
    }

    return NextResponse.json({
      data: artifact.artifact_data,
      error: null,
    });

  } catch (err) {
    console.error('Error in GET /api/creator-workflows/[id]/optimize:', err);
    return NextResponse.json(
      { data: null, error: 'Failed to fetch prediction' },
      { status: 500 }
    );
  }
}

// Helper: Calculate DPS from pack results
function calculateDPSFromPacks(pack1: any, pack3: any): number {
  if (!pack1) return 50; // Default

  // Average attribute scores (Pack 1)
  const attrScores = pack1.attribute_scores || [];
  const avgAttr = attrScores.length > 0
    ? attrScores.reduce((sum: number, a: any) => sum + (a.score || 0), 0) / attrScores.length
    : 5;

  // Hook clarity (Pack 1)
  const hookClarity = pack1.hook?.clarity_score || 5;

  // Viral mechanics strength (Pack 3)
  const mechanics = pack3?.mechanics || [];
  const avgMechanic = mechanics.length > 0
    ? mechanics.reduce((sum: number, m: any) => sum + (m.strength || 0), 0) / mechanics.length
    : 50;

  // Weighted calculation
  const dps = (avgAttr * 5) + (hookClarity * 2) + (avgMechanic * 0.3);
  return Math.min(100, Math.max(0, dps));
}

// Helper: Summarize Pack 1 for step output
function summarizePack1(pack1: any): any {
  return {
    avg_attribute_score: pack1.attribute_scores?.length > 0
      ? pack1.attribute_scores.reduce((s: number, a: any) => s + (a.score || 0), 0) / pack1.attribute_scores.length
      : null,
    hook_type: pack1.hook?.type,
    hook_clarity: pack1.hook?.clarity_score,
    pacing: pack1.pacing?.score,
    clarity: pack1.clarity?.score,
    novelty: pack1.novelty?.score,
    grader_confidence: pack1.grader_confidence,
  };
}

// Helper: Generate Gate A warnings
interface GateAWarning {
  type: 'hook' | 'proof' | 'cta' | 'general';
  severity: 'low' | 'medium' | 'high';
  message: string;
}

function generateGateAWarnings(pack1: any, pack3: any): GateAWarning[] {
  const warnings: GateAWarning[] = [];

  if (!pack1) {
    warnings.push({
      type: 'general',
      severity: 'high',
      message: 'Pack 1 analysis unavailable. Results may be incomplete.',
    });
    return warnings;
  }

  // Hook effectiveness check
  const hookClarity = pack1.hook?.clarity_score || 0;
  if (hookClarity < 6) {
    warnings.push({
      type: 'hook',
      severity: hookClarity < 4 ? 'high' : 'medium',
      message: hookClarity < 4
        ? 'Hook clarity is low (< 4/10). Consider making it more direct and compelling.'
        : 'Hook could be stronger. Consider leading with a clearer value proposition.',
    });
  }

  // Pacing check
  const pacing = pack1.pacing?.score || 0;
  if (pacing < 6) {
    warnings.push({
      type: 'general',
      severity: 'medium',
      message: 'Pacing score is below optimal. Consider tightening transitions or adding pattern interrupts.',
    });
  }

  // Clarity check
  const clarity = pack1.clarity?.score || 0;
  if (clarity < 6) {
    warnings.push({
      type: 'proof',
      severity: 'medium',
      message: 'Content clarity could be improved. Ensure your main message is easy to understand quickly.',
    });
  }

  // Check for idea legos (engagement signals)
  const legos = pack1.idea_legos || {};
  const legoCount = Object.values(legos).filter(v => v === true).length;
  if (legoCount < 2) {
    warnings.push({
      type: 'general',
      severity: 'low',
      message: 'Consider adding more engagement elements (curiosity, controversy, contrarian take, etc.)',
    });
  }

  // Pack 3 viral mechanics check
  if (pack3?.mechanics) {
    const strongMechanics = pack3.mechanics.filter((m: any) => m.strength >= 70);
    if (strongMechanics.length === 0) {
      warnings.push({
        type: 'general',
        severity: 'medium',
        message: 'No strong viral mechanics detected. Consider amplifying social currency or emotional triggers.',
      });
    }
  }

  return warnings;
}
```

```typescript
// Run prediction analysis for Optimize phase
async function runOptimizeAnalysis() {
  if (!selectedWorkflowId) {
    setOptimizeError('No workflow selected');
    return;
  }

  setIsAnalyzing(true);
  setOptimizeError(null);

  try {
    const res = await fetch(`/api/creator-workflows/${selectedWorkflowId}/optimize`, {
      method: 'POST',
    });
    const { data, error } = await res.json();

    if (error) {
      setOptimizeError(error);
    } else {
      setOptimizeResult(data);
      // Update DPS from real prediction
      setCreatorDPS(data.dps_score || 0);
    }
  } catch (err) {
    setOptimizeError('Analysis failed. Please try again.');
  } finally {
    setIsAnalyzing(false);
  }
}

// Load existing prediction when entering Optimize phase
useEffect(() => {
  if (creatorPhase === 'optimize' && selectedWorkflowId && !optimizeResult) {
    // Try to load existing prediction
    fetch(`/api/creator-workflows/${selectedWorkflowId}/optimize`)
      .then(res => res.json())
      .then(({ data, error }) => {
        if (data && !error) {
          setOptimizeResult(data);
          setCreatorDPS(data.dps_score || 0);
        }
      })
      .catch(() => {}); // Ignore errors, user can run fresh analysis
  }
}, [creatorPhase, selectedWorkflowId]);
```
